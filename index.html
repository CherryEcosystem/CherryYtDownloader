balanced ydl_opts.update({ 'concurrent_fragment_downloads': 4, 'buffersize': 1024 * 128, # 128 KB read buffer 'http_chunk_size': 5 * 1024 * 1024, # 5 MB chunks 'retries': 5, 'fragment_retries': 5, 'check_formats': False, }) #
